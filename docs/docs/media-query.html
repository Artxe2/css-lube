Media Query
<style css-lubricant="v0.7.0">
	.c\=red{color:red}
	.fs\=16px{font-size:16px}
	[class].fs\=16px\!{font-size:16px}
	/* Styles with the same prefix are grouped in the same group */
	@media (min-width:768px){
		.\@min-width\=768px\@fs\=12px{font-size:12px}
		.\@min-width\=768px\@c\=blue{color:blue}
		.\@min-width\=768px\@\:active\/fs\=12px:active{font-size:1.25em}
	}
</style>

<!-- Classes that begin with an at (@) apply media queries and use the second at to separate styles from media queries -->
<span class="c=red fs=16px @min-width=768px@fs=12px">
{ color: red }
{ font-size: 16px }
{ font-size: 12px }
</span>
<span class="c=red @min-width=768px@c=blue fs=16px! @min-width=768px@fs=12px">
{ color: red }
{ color: blue }
{ font-size: 16px }
{ font-size: 12px }
</span>
<span class="@min-width=768px@:active/fs=1.25">
:active { font-size: 1.25em }
</span>

<style css-lubricant="v0.7.0">
	@media not (min-width:768px){
		.\@\!min-width\=768px\@fs\=12px{font-size:12px}
	}
</style>

<!-- Media queries can use an exclamation point (!) to apply the not logic -->
<span class="@!min-width=768px@fs=12px">
{ font-size: 12px }
</span>

<style css-lubricant="v0.7.0">
	@container (min-width:768px){
		.\@\@container_min-width\=768px\@fs\=10px{font-size:10px}
	}
	@media screen and (min-width:768px){
		.\@screen_and_min-width\=768px\@fs\=10px{font-size:10px}
	}
</style>

<!-- If you use two at (@), the query does not have media as a prefix -->
<span class="@@container_min-width=768px@fs=10px">
{ font-size: 10px }
</span>
<span class="@screen_and_min-width=768px@fs=10px">
{ font-size: 10px }
</span>

<!-- Media queries also provide some shorthands -->
<style css-lubricant="v0.7.0">
	@media (min-width:640px) and (max-width:767px){
		.\@\!width\=640\~767px\@fs\=10px{color:red}
	}
	@media (min-width:768px){
		.\@md\@fs\=12px{color:red}
	}
	@media not (min-width:768px){
		.\@\!md\@fs\=12px{color:red}
	}
	@media (prefers-color-scheme:dark) and (min-width:768px){
		.\@\@dark_and_md\@fs\=12px{color:red}
	}
</style>

<span class="@width=640~767px@c=red">
{ color: red }
</span>
<span class="@md@c=red">
{ color: red }
</span>
<span class="@!md@c=red">
{ color: red }
</span>
<span class="@@all_and_md@c=red">
{ color: red }
</span>
